                            TalkMessage(31);
                            break;
                    }
                    goto LBL_02BF;
                }
                TalkMessage(32);
                break;
            case 2:
                TalkMessage(33);
                break;
        }
    } else {
        TalkOpen();
        TalkMessage(0);
    }
} else {
    TalkOpen();
    TalkMessage(0);
}
TalkClose();
exit;
