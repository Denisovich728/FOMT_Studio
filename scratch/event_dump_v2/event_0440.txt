        Set_Flag("Flag_0xE9", 2);
        Add_Festival_Friendship_Bonus(0x1A);
        Refresh_Map_Entities();
    } else {
        if ((Check_Flag("Flag_0xF1") == 1)) {
            Set_Flag("Flag_0xF1", 2);
            Add_Festival_Friendship_Bonus(0x1A);
            Refresh_Map_Entities();

            Give_Friendship_Points("Gray", 0x14);

    Warp_Player("Mineral_Beach", 0x54, 0x78);
    SetEntityPosition(0, 0x54, 0x78, 0);
    PlaySong(0, 0xCD);

}
