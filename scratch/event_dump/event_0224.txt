if ((Check_Flag(0x211) == 1)) {
    Fade_Out(3, 1);
    Show_Profile_UI(0);

    if ((Check_Flag(0x16) == 1)) {

            Give_Item("Jewel of Truth");
            TalkOpen();

            Set_Flag(0x211, 1);

            TalkClose();

    }

}
